THE 12H CLOCK LOGIC --- function updateClock() { const now = new Date(); // Setting hour12: true gives us that classic 12h format with AM/PM const timeString = now.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit', second: '2-digit', hour12: true }); document.getElementById('clock').innerText = timeString; } // --- 2. THE WEATHER LOGIC --- async function updateWeather() { const weatherEl = document.getElementById('weather-display'); // You can replace 'London' with your city or use geolocation! const API_KEY = 'YOUR_OPENWEATHER_API_KEY'; const CITY = 'London'; try { // If you don't have a key yet, this will catch and show demo data const response = await fetch(`https://api.openweathermap.org/data/2.5/weather?q=${CITY}&units=metric&appid=${API_KEY}`); const data = await response.json(); const temp = Math.round(data.main.temp); const condition = data.weather[0].main; weatherEl.innerHTML = `cloud ${temp}°C // ${condition}`; } catch (error) { // DEMO MODE: Shows if no API key is present weatherEl.innerHTML = `sunny 22°C // SUNNY`; } }
